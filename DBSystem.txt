CREATE DATABASE ST;
USE ST;

Create Table users(
  id int,
  fName varchar(20),
  lName varchar(20),
  pNumber varchar(20),
  email varchar(20),
  password varchar(20),
  type varchar(20)
);

Create Table orders(
  id int,
  uId int,
  employeeId int,
  orderDate Date,
  requiredDate Date,
  shippedDate Date,
  address varchar(20)
);

Create Table orderDetails(
  id int,
  userId varchar(20),
  total decimal,
  paymentId int,
  createdAt timestamp
);

Create Table orderItems(
  id int,
  orderId int,
  productId int,
  quantity int,
  discount bool,
  status varchar(20),
  createdAt timestamp
);

Create Table products(
  id int,
  name varchar(20),
  description varchar(20),
  supplierId int,
  price int,
  brandId int,
  stock int,
  categoryId varchar(20),
  discount int,	
  status varchar(20),
  pictureId int
);

Create Table cartItem(
  id int,
  productId int,
  orderId int,
  quantity int,
  createdAt timestamp
);
Create Table paymentDetail(
  id int,
  orderId int,
  amount int ,
  status varchar(20),
  createdAt timestamp
);

Create Table brands(
  id int,
  name varchar(20),
  region varchar(20)
);

Create Table supplier(
  id int,
  name varchar(20),
  pNumber int,
  email varchar(20)
);

Create Table pictures(
  id int,
  productId int,
  pictureDetail varchar(20)
);

Create Table category(
  id int,
  parentId varchar(20),
  categoryName varchar(20)
);
